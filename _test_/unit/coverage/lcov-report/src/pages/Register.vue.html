<script>
import url from "../../config/url";
import { post } from "../axios.js";
export default {
  name: "Register",
  data: () => {
    return {
      type: ["", "info", "success", "warning", "danger"],
      notifications: {
        topCenter: false
      },
      verify_code: false,
      username: "",
      email: "",
      password: "",
      confirm_password: "",
      code: "",
      users: []
    };
  },
  methods: {
    notifyVue(top_or_down, right_or_left, message, iColor) {
      var color = iColor;
 
      this.$notify({
        message: message,
        icon: "add_alert",
        horizontalAlign: right_or_left,
        verticalAlign: top_or_down,
        type: this.type[color]
      });
    },
 
    click_register: function() {
      if (
        this.username.length === 0 ||
        this.password.length === 0 ||
        this.email.length === 0
      ) {
        this.notifyVue("top", "center", "Hãy điền đầy đủ thông tin", 4);
 
        return;
      }
      /* istanbul ignore next */
      if (this.password != this.confirm_password) {
        this.notifyVue("top", "center", "Password confirm không chính xác", 4);
 
        return;
      }
      /* istanbul ignore next */
      if (!this.verify_code) {
        this.send_a_code_to_email();
        this.verify_code = true;
 
        return;
      }
      /* istanbul ignore next */
      this.create();
    },
 
    create: async function() {
      let user = {
        email: this.email,
        code: this.code
      };
      const api = url.url + "/api/logup/";
      const data = await post(api, user);
 
      /* istanbul ignore else */
      if (data.success) {
        this.notifyVue("top", "center", "Tạo tài khoản thành công", 2);
        /* istanbul ignore next */
        this.$router.push("/login");
      } else if (data.success === false) {
        this.notifyVue("top", "center", "Code không chính xác", 4);
        this.code = "";
      } else {
        this.notifyVue("top", "center", "this.$t('common.serverError')", 4);
      }
    },
 
    send_a_code_to_email: async function() {
      let user = {
        username: this.username,
        password: this.password,
        email: this.email
      };
      const api = url.url + "/api/logup/email";
      const data = await post(api, user);
      /* istanbul ignore else */
 
      if (!data.success && data.active) {
        this.notifyVue("top", "center", data.message, 3);
      } else if (!data.success && !data.active) {
        this.notifyVue("top", "center", data.message, 4);
        this.verify_code = true;
      } else if (data.success) {
        this.notifyVue(
          "top",
          "center",
          "Chúng tôi đã gửi 1 code tới email của bạn. Hãy kiểm tra.",
          2
        );
        this.verify_code = true;
      } else {
        this.notifyVue("top", "center", "this.$t('common.serverError')", 4);
      }
    }
  }
};
</script>
