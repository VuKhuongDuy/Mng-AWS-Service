export default {
  name: "Login",
 
  data: () => {
    return {
      type: ["", "info", "success", "warning", "danger"],
      notifications: {
        topCenter: false
      },
      username: "",
      password: "",
      show: false,
      title: "Thông báo"
    };
  },
 
  methods: {
    notifyVue(top_or_down, right_or_left, message, iColor) {
      var color = iColor;
 
      this.$notify({
        message: message,
        icon: "add_alert",
        horizontalAlign: right_or_left,
        verticalAlign: top_or_down,
        type: this.type[color]
      });
    },
 
    click_login: async function() {
      const api = url.url + "/api/login";
 
      let user = {
        username: this.username,
        password: this.password
      };
 
      const data = await post(api, user);
      
      if (data.success) {
      /* istanbul ignore else */
      console.log(data)
        document.cookie = escape("username") + "=" + escape(this.username) + ";"
